\begin{questions}
	\question Expand and simplify the following:
	\begin{parts}
		\part $(x+1)(x+2)(x+3)$
		\begin{solution}
			$$x^3+6x^2+15x+18$$
		\end{solution}
		
		\part $(x+3)^3$
		\begin{solution}
			$$x^3+9x^2+27x+27$$
		\end{solution}
		
		\part $(x-2)(x-3)(x-4)$
		\begin{solution}
			$$x^3-9x^2+26x-24$$
		\end{solution}
		
		\part $x(x-2)(x+2)$
		\begin{solution}
			$$x^3-4x$$
		\end{solution}
		
		\part $(2-x)(3-x)(4-x)$
		\begin{solution}
			$$-x^3+9x^2-26x+24$$
		\end{solution}
		
		\part $(y+1)(4-y)(3+y)$
		\begin{solution}
			$$-y^3+13y+12$$
		\end{solution}
		
		\part $(a+b)(b+c)(a+c)$
		\begin{solution}
			$$a^2b+a^2c+ac^2+ab^2+b^2c+bc^2+2abc$$
		\end{solution}
	\end{parts}
	
	\question The base and height of a rectangle is $x+3$ and $3x-2$.
	\begin{parts}
		\part Find an expression for the perimeter of the rectangle.
		\begin{solution}
			$$2(x+3 + 3x-2)\equiv8x+2$$
		\end{solution}
		
		\part Find an expression for the area of the rectangle.
		\begin{solution}
			$$3x^2+7x-6$$
		\end{solution}
	\end{parts}
	
	\question The length, width and height of a cuboid are $x+1$, $2x-2$ and $10-x$.
	\begin{parts}
		\part Find an expression for the volume of the cuboid. Expand and simplify your answer.
		\begin{solution}
			$$-2x^3+20x^2+2x-20$$
		\end{solution}
		
		\part Find an expression for the surface area of the cuboid. Expand and simplify your answer.
		\begin{solution}
			$$2\big[(x+1)(2x-2)+(2x-2)(10-x)+(x+1)(10-x)\big]\equiv-x^2+31x-12$$
		\end{solution}
	\end{parts}
	
	\question Find the area and circumference of a circle with radius $x+2$. Expand and simplify your answer.
	\begin{solution}
		\centering
			$C=2\pi x+ 4\pi$ and $A=\pi x^2 + 2\pi x+ 2\pi$.
	\end{solution}
	
\end{questions}
